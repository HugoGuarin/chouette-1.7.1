package fr.certu.chouette.service.importateur.multilignes.pegase;

public class ItineraireException extends Exception {
	
	public ItineraireException() {
		super();
	}
	
	public ItineraireException(String message) {
		super(message);
	}
	
	public ItineraireException(String message, Throwable cause) {
		super(message, cause);
	}
	
	public ItineraireException(Throwable cause) {
		super(cause);
	}
}
